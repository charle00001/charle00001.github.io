关于cookie的一个实践

在网页中使用 JavaScript 和 Cookie 来创建一个允许用户选择是否接受 Cookie 的弹窗 + JavaScript 代码示例

刚刚弹窗的代码：#

<script>
    function showDisclaimer() {
        // 检查是否存在名为 "disclaimer" 的 Cookie
        var disclaimerCookie = getCookie("disclaimer");
        if (disclaimerCookie === "同意") {
            // 如果存在 "disclaimer" Cookie，并且值为 "同意"，则不显示弹窗
            return;
        }

        // 显示弹窗
        var disclaimer = "这是一个测试弹窗，当你同意cookie后不会再显示对话框.";
        window.alert(disclaimer);

        // 询问用户是否同意记录cookie
        var result = window.confirm("您是否同意记录cookie？（本站使用cookie来测试代码）");
        if (result) {
            // 如果用户点击确认，将 "disclaimer" Cookie 的值设置为 "同意"
            setCookie("disclaimer", "同意");
            // 显示已同意的提示消息
            window.alert("您已同意使用cookie.");
        } else {
            // 如果用户点击取消或关闭对话框，执行重定向到浏览器空白页（可根据需求进行修改）
            window.location.href = "about:blank";
        }
    }

    // 获取指定名称的 Cookie 的值
    function getCookie(name) {
        var cookies = document.cookie.split("; ");
        for (var i = 0; i < cookies.length; i++) {
            var cookie = cookies[i].split("=");
            if (cookie[0] === name) {
                return cookie[1];
            }
        }
        return "";
    }

    // 设置 Cookie
    function setCookie(name, value) {
        var expires = new Date();
        expires.setFullYear(expires.getFullYear() + 1); // 设置 Cookie 有效期为一年
        document.cookie = name + "=" + value + "; expires=" + expires.toUTCString();
    }

    // 在适当的时候调用 showDisclaimer() 函数，例如在页面加载完成后或点击某个按钮时
    showDisclaimer();
</script>

可以用来做一个简易的免责声明、或者“同意 or No”的弹窗，不同意就给你重定向到about:blank（空白页）

也可以做一些别的操作，只要是 JavaScript 能做到的都行

缺点是如果你禁用了 JS，那这代码就没用
